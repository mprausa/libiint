    \begin{align*}
        f(x)g(x)
        &=
        \sum\limits_{n=n_0}^\infty
        \sum\limits_{k=k_0}^\infty
        a_n
        b_k
        x^{n+k}
        \\
        &=
        \sum\limits_{n=n_0}^\infty
        \sum\limits_{k=k_0+n}^\infty
        a_n
        b_{k-n}
        x^k
        \\
        &=
        \sum\limits_{k=k_0+n_0}^\infty
        x^k
        \sum\limits_{n=n_0}^{k-k_0}
        a_n
        b_{k-n}
    \end{align*}
