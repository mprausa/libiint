        \omega_a(x) &= \frac1{x-a}
    \end{align*}

    \begin{align*}
        \omega_{a\neq x}(x+\delta)
        &=
        \frac1{(x-a)+\delta}
        \\
        &=
        \frac1{x-a}
        \frac1{1+\frac{\delta}{x-a}}
        \\
        &=
        \frac1{x-a}
        \sum\limits_{k=0}^\infty
        \left(\frac{\delta}{a-x}\right)^k
        \\
        &=
        -
        \sum\limits_{k=0}^\infty
        \delta^k
        \frac{1}{(a-x)^{k+1}}
    \end{align*}

    \begin{align*}
        \omega_x(x+\delta)
        &=
        \frac1{\delta}
    \end{align*}

    \begin{align*}
        a_{2k\geq0}(\omega_{a\neq x};x) &= -\frac{1}{(a-x)^{k+1}}
        \\
        a_{-2}(\omega_x;x) &= 1
    \end{align*}
